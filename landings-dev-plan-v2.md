```tsx
// components/ui/Avatar.tsx
'use client';

import { useState } from 'react';
import Image from 'next/image';

interface AvatarProps {
  src: string;
  name: string;
  size?: number;
}

export function Avatar({ src, name, size = 48 }: AvatarProps) {
  const [error, setError] = useState(false);
  
  const fallbackUrl = `https://ui-avatars.com/api/?name=${encodeURIComponent(name)}&background=C9A227&color=050505&size=${size * 2}&bold=true`;
  
  return (
    <div 
      className="relative rounded-full overflow-hidden ring-2 ring-gold/30"
      style={{ width: size, height: size }}
    >
      <Image
        src={error ? fallbackUrl : src}
        alt={name}
        fill
        className="object-cover"
        onError={() => setError(true)}
      />
    </div>
  );
}
```

### React Hook Implementation

```typescript
// hooks/useScrollAnimation.ts
import { useEffect, useRef, useState } from 'react';

interface UseScrollAnimationOptions {
  threshold?: number;
  rootMargin?: string;
  triggerOnce?: boolean;
}

export function useScrollAnimation(options: UseScrollAnimationOptions = {}) {
  const {
    threshold = 0.1,
    rootMargin = '0px 0px -50px 0px',
    triggerOnce = true
  } = options;

  const ref = useRef<HTMLDivElement>(null);
  const [isInView, setIsInView] = useState(false);

  useEffect(() => {
    const element = ref.current;
    if (!element) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
          if (triggerOnce) {
            observer.unobserve(element);
          }
        } else if (!triggerOnce) {
          setIsInView(false);
        }
      },
      { threshold, rootMargin }
    );

    observer.observe(element);
    return () => observer.disconnect();
  }, [threshold, rootMargin, triggerOnce]);

  return { ref, isInView };
}
```

```tsx
// components/ui/InfiniteMarquee.tsx
'use client';

import { cn } from '@/lib/utils';
import { useEffect, useRef, useState } from 'react';

interface InfiniteMarqueeProps {
  children: React.ReactNode;
  className?: string;
  speed?: number; // seconds for one loop
  direction?: 'left' | 'right';
  pauseOnHover?: boolean;
  gap?: number; // gap between items in px
}

export function InfiniteMarquee({
  children,
  className,
  speed = 40,
  direction = 'left',
  pauseOnHover = true,
  gap = 48,
}: InfiniteMarqueeProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [duplicateCount, setDuplicateCount] = useState(2);

  // Calculate how many duplicates needed to fill screen
  useEffect(() => {
    const calculateDuplicates = () => {
      if (containerRef.current) {
        const containerWidth = containerRef.current.offsetWidth;
        const contentWidth = containerRef.current.scrollWidth / duplicateCount;
        const needed = Math.ceil((containerWidth * 2) / contentWidth) + 1;
        setDuplicateCount(Math.max(2, needed));
      }
    };

    calculateDuplicates();
    window.addEventListener('resize', calculateDuplicates);
    return () => window.removeEventListener('resize', calculateDuplicates);
  }, [children, duplicateCount]);

  return (
    <div
      ref={containerRef}
      className={cn('marquee-container', className)}
    >
      <div
        className={cn(
          'marquee-track',
          direction === 'right' && 'reverse',
          pauseOnHover && 'hover:pause'
        )}
        style={{
          animationDuration: `${speed}s`,
          gap: `${gap}px`,
        }}
      >
        {/* Original content */}
        {children}
        {/* Duplicate for seamless loop */}
        {children}
      </div>
    </div>
  );
}
```

```typescript
        return new Promise<HTMLImageElement>((resolve, reject) => {
          const img = new Image();
          
          // Generate filename based on pattern
          const frameNumber = paddingDigits > 0
            ? String(i + 1).padStart(paddingDigits, '0')
            : String(i + 1);
          
          img.src = `${basePath}/frame_${frameNumber}.webp`;
          
          img.onload = () => {
            loadedCount++;
            if (isMounted) {
              setLoadProgress(loadedCount / frameCount);
            }
            loadedImages[i] = img;
            resolve(img);
          };
          
          img.onerror = () => {
            reject(new Error(`Failed to load frame ${i + 1}`));
          };
        });
      });

      try {
        await Promise.all(imagePromises);
        if (isMounted) {
          setImages(loadedImages);
          setIsLoaded(true);
        }
      } catch (err) {
        if (isMounted) {
          setError(err instanceof Error ? err.message : 'Failed to load images');
        }
      }
    };

    loadImages();

    return () => {
      isMounted = false;
    };
  }, [frameCount, basePath, paddingDigits]);

  return { images, isLoaded, loadProgress, error };
}
```

### 4. SCROLL LOGIC

```tsx
import { useScroll, useMotionValueEvent } from 'framer-motion';

// Inside component
const containerRef = useRef<HTMLDivElement>(null);
const canvasRef = useRef<HTMLCanvasElement>(null);
const [currentFrameIndex, setCurrentFrameIndex] = useState(0);

// Frame count from questionnaire (C1)
const FRAME_COUNT = 70; // Dynamic from config

// Setup scroll tracking
const { scrollYProgress } = useScroll({
  target: containerRef,
  offset: ["start start", "end end"]
});

// Draw frame function
const drawFrame = useCallback((frameIndex: number) => {
  const canvas = canvasRef.current;
  const ctx = canvas?.getContext('2d');
  const img = images[frameIndex];

  if (ctx && img && canvas) {
    // Clear canvas
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // Calculate contain-fit positioning
    const canvasAspect = canvas.width / canvas.height;
    const imgAspect = img.width / img.height;

    let drawWidth, drawHeight, drawX, drawY;

    if (imgAspect > canvasAspect) {
      // Image is wider than canvas
      drawWidth = canvas.width;
      drawHeight = canvas.width / imgAspect;
      drawX = 0;
      drawY = (canvas.height - drawHeight) / 2;
    } else {
      // Image is taller than canvas
      drawHeight = canvas.height;
      drawWidth = canvas.height * imgAspect;
      drawX = (canvas.width - drawWidth) / 2;
      drawY = 0;
    }

    ctx.drawImage(img, drawX, drawY, drawWidth, drawHeight);
  }
}, [images]);

// Listen to scroll and update frame
useMotionValueEvent(scrollYProgress, "change", (progress) => {
  if (!isLoaded) return;

  const frameIndex = Math.min(
    Math.floor(progress * FRAME_COUNT),
    FRAME_COUNT - 1
  );

  if (frameIndex !== currentFrameIndex) {
    setCurrentFrameIndex(frameIndex);
    drawFrame(frameIndex);
  }
});

// Initial draw when loaded
useEffect(() => {
  if (isLoaded && images.length > 0) {
    drawFrame(0);
  }
}, [isLoaded, images, drawFrame]);
```

```tsx
// components/sections/02-HeroScrollAnimation.tsx
'use client';

import { useRef, useState, useCallback, useEffect } from 'react';
import { motion, useScroll, useTransform, useMotionValueEvent } from 'framer-motion';
import { useImageSequence } from '@/hooks/useImageSequence';
import { heroTextSections } from '@/config/heroContent';
import { renderTextSection } from './TextSections';

export function HeroScrollAnimation() {
  const containerRef = useRef<HTMLDivElement>(null);
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [currentFrameIndex, setCurrentFrameIndex] = useState(0);

  // Configuration from questionnaire
  const FRAME_COUNT = 140;  // [C1]
  const PADDING_DIGITS = 3; // [C2]

  // Load images
  const { images, isLoaded, loadProgress, error } = useImageSequence({
    frameCount: FRAME_COUNT,
    basePath: '/scroll-frames',
    filePattern: 'frame_{n}.webp',
    paddingDigits: PADDING_DIGITS,
  });

  // Scroll progress
  const { scrollYProgress } = useScroll({
    target: containerRef,
    offset: ['start start', 'end end'],
  });

  // Draw frame to canvas
  const drawFrame = useCallback((frameIndex: number) => {
    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    const img = images[frameIndex];

    if (ctx && img && canvas) {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Calculate contain-fit
      const scale = Math.min(
        canvas.width / img.width,
        canvas.height / img.height
      );
      const x = (canvas.width - img.width * scale) / 2;
      const y = (canvas.height - img.height * scale) / 2;

      ctx.drawImage(img, x, y, img.width * scale, img.height * scale);
    }
  }, [images]);

  // Handle resize
  useEffect(() => {
    const handleResize = () => {
      const canvas = canvasRef.current;
      if (canvas) {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        if (images[currentFrameIndex]) {
          drawFrame(currentFrameIndex);
        }
      }
    };

    handleResize();
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, [currentFrameIndex, images, drawFrame]);

  // Update frame on scroll
  useMotionValueEvent(scrollYProgress, 'change', (progress) => {
    if (!isLoaded) return;

    const frameIndex = Math.min(
      Math.floor(progress * FRAME_COUNT),
      FRAME_COUNT - 1
    );

    if (frameIndex !== currentFrameIndex) {
      setCurrentFrameIndex(frameIndex);
      drawFrame(frameIndex);
    }
  });

  // Initial draw
  useEffect(() => {
    if (isLoaded && images.length > 0) {
      drawFrame(0);
    }
  }, [isLoaded, images, drawFrame]);

  return (
    <>
      {/* Loading Screen */}
      {!isLoaded && (
        <div className="fixed inset-0 bg-[#050505] z-50 flex items-center justify-center">
          <div className="text-center">
            <div className="relative w-24 h-24 mx-auto mb-6">
              <svg className="w-full h-full -rotate-90">
                <circle
                  cx="48" cy="48" r="44"
                  stroke="rgba(255,255,255,0.1)"
                  strokeWidth="4"
                  fill="none"
                />
                <circle
                  cx="48" cy="48" r="44"
                  stroke="#C9A227"
                  strokeWidth="4"
                  fill="none"
                  strokeDasharray={276}
                  strokeDashoffset={276 - (276 * loadProgress)}
                  strokeLinecap="round"
                />
              </svg>
              <span className="absolute inset-0 flex items-center justify-center text-white/60 text-sm">
                {Math.round(loadProgress * 100)}%
              </span>
            </div>
            <p className="text-white/40 text-sm tracking-widest uppercase">
              Loading Experience
            </p>
          </div>
        </div>
      )}

      {/* Hero Section */}
      <section
        ref={containerRef}
        className="relative h-[500vh] bg-[#050505]"
      >
        {/* Sticky Canvas Container */}
        <div className="sticky top-0 h-screen w-full">
          {/* Canvas */}
          <canvas
            ref={canvasRef}
            className="absolute inset-0 w-full h-full"
          />

          {/* Text Overlays */}
          {heroTextSections.map((section) => {
            const opacity = useTransform(
              scrollYProgress,
              [
                section.startProgress,
                section.startProgress + 0.03,
                section.endProgress - 0.03,
                section.endProgress,
              ],
              [0, 1, 1, 0]
            );

            const positionClasses = {
              center: 'items-center justify-center',
              left: 'items-center justify-start pl-8 md:pl-16 lg:pl-24',
              right: 'items-center justify-end pr-8 md:pr-16 lg:pr-24',
            };

            return (
              <motion.div
                key={section.id}
                className={`absolute inset-0 flex ${positionClasses[section.position]} pointer-events-none z-10`}
                style={{ opacity }}
              >
                <div className="pointer-events-auto">
                  {renderTextSection(section)}
                </div>
              </motion.div>
            );
          })}
        </div>
      </section>
    </>
  );
}
```
